const mhs = [‘Sandika’,Doddy’,’Erik’]

//looping biasa

for (let i = 0; i < mhs.length; i++) {

       console.log (mhs[i]);

}

//looping dengan forEach

mhs.forEach( m => console.log (m));

jika ingin mendapatkan indeksnya:

mhs.forEach( (m,i) => console.log (`${m} adalah mahasiswa ke-${i+1}`);

forEach juga bisa digunakan untuk melooping NodeList

 

1.       FOR..OF  à CREATE A LOOP ITERATING OVER ITERABLE OBJECT

Iterable Object:

a.       String

const nama = ‘Sandhika’;

for (const n of nama) {

       console.log (n);

}

b.      Array

const mhs = [‘Sandhika’,’Doddy’,’Erik’];

for (const m of mhs) {

       console.log (m);

}

jika ingin mendapatkan indeksnya:

for (const [i,m] of mhs.entries()) {

       console.log (`${m} adalah mahasiswa ke-${i+1}`);

}

c.        NodeList

cons liNama = document.querySelectorAll(‘.nama’);

for (n of liNama) {console.log (n.innerHTML);}

d.      Arguments

function jumlahkanAngka() {

       let jumlah = 0;

       for (a of arguments) {jumlah += a;}

return jumlah;

}

console.log (jumlahkanAngka(1,2,3,4,5));

f.        Map

g.       Set

h.      User-defined Iterable
